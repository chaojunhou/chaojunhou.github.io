class Solution:
    # @param {string} s
    # @return {string}
    def shortestPalindrome(self, s):
        haystack = s[::-1]
        m = len(s)
        j = -1
        prefix=self.compute_prefix(s)
        for i in range(m):
            while j > -1 and s[j+1]!=haystack[i]:
                j = prefix[j]
            if s[j+1] == haystack[i]:
                j = j+1
        return s[m-1:j:-1]+s
    def compute_prefix(self,P):
        m=len(P)
        prefix=[-1 for i in range(m)]
        j = -1
        for i in range(1,m):
            while j > -1 and P[j+1]!=P[i]:
                j=prefix[j]
            if P[j+1]==P[i]:
                j+=1
            prefix[i]=j
        return prefix
